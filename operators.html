JavaScript Operators

JavaScript provides a variety of operators to perform operations on variables and values. Here are some common types of operators:

Arithmetic Operators

  • +: Addition
  • -: Subtraction
  • *: Multiplication
  • /: Division
  • %: Modulus (remainder)
  • **: Exponentiation (ES2016)

Comparison Operators

  • ==: Equal to (value only)
  • ===: Strict equal to (value and type)
  • !=: Not equal to (value only)
  • !==: Strict not equal to (value and type)
  • <: Less than
  • >: Greater than
  • <=: Less than or equal to
  • >=: Greater than or equal to

Logical Operators

  • &&: Logical AND
  • ||: Logical OR
  • !: Logical NOT

Assignment Operators

  • =: Assignment
  • +=: Add and assign
  • -=: Subtract and assign
  • *=: Multiply and assign
  • /=: Divide and assign
  • %=: Modulus and assign

Bitwise Operators

  • &: Bitwise AND
  • |: Bitwise OR
  • ^: Bitwise XOR
  • ~: Bitwise NOT
  • <<: Left shift
  • >>: Right shift
  • >>>: Unsigned right shift

Type Operators

  • typeof: Returns the type of a variable
  • instanceof: Tests whether an object is an instance of a specific class or constructor function

Conditional (Ternary) Operator

The conditional operator is a shorthand for an if-else statement:

  • condition ? expr1 : expr2: If the condition is true, it evaluates to expr1; otherwise, it evaluates to expr2.

Example


        let a = 10;
        let b = 5;

        // Arithmetic Operators
        console.log(a + b); // Output: 15
        console.log(a - b); // Output: 5
        console.log(a * b); // Output: 50
        console.log(a / b); // Output: 2
        console.log(a % b); // Output: 0

        // Comparison Operators
        console.log(a == b); // Output: false
        console.log(a === 10); // Output: true
        console.log(a != b); // Output: true
        console.log(a !== '10'); // Output: true
        console.log(a > b); // Output: true

        // Logical Operators
        console.log(true && false); // Output: false
        console.log(true || false); // Output: true
        console.log(!true); // Output: false

        // Assignment Operators
        a += 5; // a = a + 5
        console.log(a); // Output: 15
        b *= 2; // b = b * 2
        console.log(b); // Output: 10
        b -= 5; // b = b - 5
        console.log(b); // Output: 5
        b /= 5; // b = b / 5
        console.log(b); // Output: 1
        b %= 2; // b = b % 2
        console.log(b); // Output: 1

        // Bitwise Operators
        console.log(a & b); // Output: 10 (1010 & 1010)
        console.log(a | b); // Output: 15 (1111 | 1010)
        console.log(a ^ b); // Output: 5 (1111 ^ 1010)
        console.log(~a); // Output: -16 (bitwise NOT)
        console.log(a << 1); // Output: 30 (left shift)
        console.log(a >> 1); // Output: 7 (right shift)

        // Type Operators
        console.log(typeof b); // Output: "number"
        console.log(typeof 'Hello'); // Output: "string"    
        console.log(typeof true); // Output: "boolean"
        console.log(typeof null); // Output: "object" (this is a known JavaScript quirk)
        console.log(typeof undefined); // Output: "undefined"
        console.log(typeof Symbol('symbol')); // Output: "symbol"
        console.log(typeof [1, 2, 3]); // Output: "object" (arrays are objects in JavaScript)
        console.log(typeof function() {}); // Output: "function" (functions are a special type of object)
        console.log(typeof new Date()); // Output: "object" (Date is an object in JavaScript)
        console.log(typeof NaN); // Output: "number" (NaN is a special numeric value)
        console.log(typeof BigInt(12345678901234567890)); // Output: "bigint" (BigInt is a new numeric type in JavaScript)
        console.log(a instanceof Number); // Output: false
        console.log(b instanceof Number); // Output: false
        console.log(b instanceof Object); // Output: false       

        // Conditional (Ternary) Operator
        let max = (a > b) ? a : b;
        console.log(max); // Output: 15
        // Example of a function using the conditional operator
        function checkEvenOdd(num) {
            return (num % 2 === 0) ? 'Even' : 'Odd';
        }
        console.log(checkEvenOdd(10)); // Output: "Even"
        console.log(checkEvenOdd(7)); // Output: "Odd"
